18 package com.agorapulse.micronaut.snitch;
19 
20 import io.micronaut.http.HttpStatus;
21 import io.micronaut.http.client.exceptions.HttpClientResponseException;
22 import org.slf4j.Logger;
23 import org.slf4j.LoggerFactory;
24 
25 /**
26  * Default snitch service uses the SnitchClient to report to the snitch service.
27  *
28  * Successful reports are send at most once per 5 minutes.
29  */
30 public class DefaultSnitchService implements SnitchService {
31 
32     public static final String OK = "Got it, thanks!";
33 
34     private static final Logger LOGGER = LoggerFactory.getLogger(SnitchService.class);
35     private static final long CALL_INTERVAL = 5L * 60 * 1000; // 5 minutes
36 
37     private final SnitchClient client;
38     private final SnitchJobConfiguration configuration;
39 
40     private long lastSuccessfulCallTime;
41     private long lastTooManyRequestsCallTime;
42 
43     public DefaultSnitchService(SnitchClient client, SnitchJobConfiguration configuration) {
44         this.client = client;
45         this.configuration = configuration;
46     }
47 
48     @Override
49     public boolean snitch(boolean success) {
50         if (isRequestAlreadyDoneRecently(lastSuccessfulCallTime) || isRequestAlreadyDoneRecently(lastTooManyRequestsCallTime)) {
51             return true;
52         }
53         try {
54             boolean lastSuccessfulCall = OK.equals(client.snitch(configuration.getId(), success ? "0" : "1"));
55 
56             if (lastSuccessfulCall) {
57                 lastSuccessfulCallTime = System.currentTimeMillis();
58             }
59 
60             return lastSuccessfulCall;
61         } catch (HttpClientResponseException ex) {
62             if (ex.getStatus() == HttpStatus.TOO_MANY_REQUESTS) {
63                 this.lastTooManyRequestsCallTime = System.currentTimeMillis();
64             } else {
65                 LOGGER.warn("Exception notifying snitch " + configuration.getName(), ex);
66             }
67             return false;
68         } catch (Exception ex) {
69             LOGGER.warn("Exception notifying snitch " + configuration.getName(), ex);
70             return false;
71         }
72     }
73 
74     private boolean isRequestAlreadyDoneRecently(long callTime) {
75         return callTime > 0 && callTime + CALL_INTERVAL > System.currentTimeMillis();
76     }
77 }

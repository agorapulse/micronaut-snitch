18  package com.agorapulse.micronaut.snitch;
19  
20  import io.micronaut.aop.MethodInterceptor;
21  import io.micronaut.aop.MethodInvocationContext;
22  import io.micronaut.context.BeanContext;
23  import io.micronaut.inject.qualifiers.Qualifiers;
24  import javax.inject.Singleton;
25  
26  /**
27   * Interceptor for @{@link Snitch} annotation.
28   */
29  @Singleton
30  public class SnitchInterceptor implements MethodInterceptor<Object, Object> {
31  
32      private final BeanContext beanContext;
33  
34      public SnitchInterceptor(BeanContext beanContext) {
35          this.beanContext = beanContext;
36      }
37  
38      @Override
39      public Object intercept(MethodInvocationContext<Object, Object> methodContext) {
40          String configurationName = methodContext.getValue(Snitch.class, String.class).orElse(Snitch.DEFAULT_CONFIGURATION_NAME);
41          boolean before = methodContext.getValue(Snitch.class, "before", Boolean.class).orElse(false);
42          SnitchService service = beanContext.findBean(SnitchService.class, Qualifiers.byName(configurationName))
43                  .orElseGet(() -> beanContext.getBean(SnitchService.class));
44          try {
45              if (before) {
46                  service.snitch();
47              }
48  
49              Object result = methodContext.proceed();
50  
51              if (!before) {
52                  service.snitch();
53              }
54  
55              return result;
56          } catch (Throwable e) {
57              if (!before) {
58                  service.snitch(false);
59              }
60  
61              throw e;
62          }
63      }
64  
65  }
